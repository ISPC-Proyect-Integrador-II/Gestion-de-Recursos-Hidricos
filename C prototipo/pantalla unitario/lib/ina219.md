#include <Arduino.h>    
#include <Wire.h>
#include <Adafruit_INA219.h>

#define EEPROM_I2C_ADDRESS 0x40

// Inicializar el sensor INA219x
Adafruit_INA219 ina219;

void setup() {
    Serial.begin(9600);
    Wire.begin();
    
    Serial.println("üîç Inicializando sensor INA219...");
    
    if (!ina219.begin()) {
        Serial.println(" Error: No se detect√≥ el sensor INA219.");
        while (1);
    }
}

void loop() {
    float voltaje = ina219.getBusVoltage_V();  // Leer voltaje del bus DC
    float corriente = ina219.getCurrent_mA(); // Leer corriente en miliamperios
    float potencia = voltaje * (corriente / 1000.0); // Calcular potencia en Watts

    Serial.print(" Voltaje: ");
    Serial.print(voltaje);
    Serial.println(" V");

    Serial.print(" Corriente: ");
    Serial.print(corriente);
    Serial.println(" mA");

    Serial.print(" Potencia: ");
    Serial.print(potencia);
    Serial.println(" W");

    delay(2000);
}
